// With the addition on filters and alerts, the processing of unread counts
    // can take longer than before. The purpose of this script is to try to
    // preload the unread count if the user has a valid session before we load
    // the scripts and CSS included in this page.
    //
    // IMPORTANT: We are using a global variable for this isolated piece of
    // script to communicate with the CloudUtils library which is loading
    // the user context which includes the unread counts.
    //
    window.feedlyFasterLoad = {
      inProgress: false,
      data: null,
      onReady: null
    }

    var fasterLoad = window.feedlyFasterLoad;

    // Do we have a valid session?
    var sessionData = _fasterLoadGetCookie('feedly.session')
    var session = sessionData ? JSON.parse(sessionData) : null;

    if (
      window.location.host === 'feedly.com'
      && session
      && session.feedlyToken
      && session.feedlyExpirationTime
      && session.feedlyExpirationTime > new Date().getTime() ){
      fasterLoad.inProgress = true;

      // Get unreadCounts using existing valid session before the rest of
      // the JS and CSS content is loaded.
      var url = 'https://feedly.com/v3/markers/counts?ct=feedly.desktop&ck='
        + new Date().getTime() + '&feedlyToken=' + session.feedlyToken;

      _fasterLoadGet(url, function(data) {
        // We were able to successfully extra the unread count information.
        // If there is a listener, notify him
        fasterLoad.data = data;
        fasterLoad.inProgress = false;
        if (fasterLoad.onReady) {
          fasterLoad.onReady();
          fasterLoad.onReady = null;
        }
      }, function() {
        // We failed to load the unreadCounts information. If there is a
        // listener, notify him.
        fasterLoad.data = null;
        fasterLoad.inProgress = false;
        if (fasterLoad.onReady) {
          fasterLoad.onReady();
          fasterLoad.onReady = null;
        }
      });
    }

    // Extra named cookie. Example: feedly.session
    function _fasterLoadGetCookie(name) {
      var nameEQ = name + '=';
      var ca = document.cookie.split(';');
      for (var i = 0; i < ca.length; i++) {
        var c = ca[i];
        while (c.charAt(0) === ' ')
          c = c.substring(1, c.length);
        if (c.indexOf(nameEQ) === 0)
          return c.substring(nameEQ.length, c.length);
      }
      return null;
    }

    function _fasterLoadGet(url, onSuccess, onError) {
      var xhttp = new XMLHttpRequest();
      xhttp.onreadystatechange = function() {
        if (this.readyState === 4) {
          if (this.status === 200) {
            onSuccess(xhttp.responseText);
          } else {
            onError();
          }
        }
      }
      xhttp.open('GET', url, true);
      xhttp.send();
    }

var feedlyT0 = new Date().getTime();

    function fromFeedlyT0(){
      return new Date().getTime() - feedlyT0;

    function getFeedlyCookie()
    {
      try
      {
        var nameEQ = "feedly.session=";
        var ca = document.cookie.split(';');
        for(var i=0;i < ca.length;i++) {
          var c = ca[i];
          while (c.charAt(0)==' ') c = c.substring(1,c.length);
          if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length,c.length);
        }
        return null;
      }
      catch( e )
      {
        return null;
      }
    }

    window.errorTimer = window.setTimeout( function(){
        // The page has successfully loaded.
        var blockedElemend = document.getElementById( "feedlyBlocked" );
        if( blockedElemend == null )
          return;

        handleError()
      },
      15000
    );

    function handleError()
    {
      if( window.errorTimer != null )
      {
        window.clearTimeout( window.errorTimer );
        window.errorTimer = null;
      }

      // if loading from http, then redirect to https to avoid CDN
      if( document.location.protocol == "http" )
      {
        document.location = "https://feedly.com/";
        return;
      }
      else
      {
        // The page has successfully loaded.
        var blockedElemend = document.getElementById( "feedlyBlocked" );
        if( blockedElemend != null )
          blockedElemend.style.opacity = 1;

        document.getElementById( "feedlyStaticProgressBar" ).style.display = "none";
        document.getElementById( "feedlySplashscreen" ).style.display = "none";
      }
    }

    try
    {
      // MAPPING VERSION 4
      //
      // if this is an old URL, we should redirect it to the new URL format
      // OLD http://feedly.com/beta#welcome
      // NEW http://feedly.com/b/welcome

      var h = window.location.hash;             // example #welcome
      if( h.indexOf( "#" ) == 0 )
        h = h.slice( 1 );

      var p = window.location.pathname;         // example / /beta /dev.html /index.html

      var host = window.location.host;          // example feedly.com or cloud.feedly.com

      // CASE 0 - Old hosts (example: cloud.feedly.com or www.feedly.com)
      if( host != "feedly.com" && host != "sandbox7.feedly.com" && !host.startsWith("localhost") )
      {
        window.location.href = "https://feedly.com" + p + h;
      }
      else if( p == "/home" )
      {
        window.location.href = "https//feedly.com/i/" + h;
      }
      // CASE 1 - We have an old hash URL
      else if( h.length > 0 )
      {
        console.log( "[feedly] need to map hash:" + h + " pathname:" + p );
        switch( p )
        {
          case "/beta":
            window.location.href = "https://feedly.com/b/" + h;
            break;

          case "/alpha.html":
            window.location.href = "https//feedly.com/a/" + h;
            break;

          case "/dev.html":
            window.location.href = "https://feedly.com/d/" + h;
            break;

          case "/home":
          case "/index.html":
          case "/":
            window.location.href = "https://feedly.com/i/" + h;
            break;

          default:
          // We must be in a case where
          // http://feedly.com/i/subscription/feed/....
          // where the feed id has a hash.
        }
      }
      else if( p.indexOf( ".html" ) > 0 )
      {
        // CASE 2 - Old pathnames
        switch( p )
        {
          case "/alpha.html":
            window.location.href = "https://feedly.com/a";
            break;

          case "/dev.html":
            window.location.href = "https://feedly.com/d";
            break;

          case "/index.html":
            window.location.href = "https://feedly.com/";
            break;

          default:
        }
      }

    }
    catch( e )
    {
      // ignore
      console.log( "[feedly] url mapping failed:" + e.name + " -- " + e.message )
      window.location.href = "https://feedly.com/";
    }

    var feedlyProgressCounter  = 0;
    var feedlyProgressDuration = 1000;
    var feedlyProgressSteps	   = 50;
    var feedlyProgressInterval = window.setInterval( function(){

        if( window.location.pathname.indexOf( "/pro" ) == 2 )
        {
          window.clearInterval( feedlyProgressInterval );
          return;
        }

        feedlyProgressCounter++;

        if( feedlyProgressCounter == 1 )
        {
          var pbElem = document.getElementById( "feedlyStaticProgressBar" );
          if( pbElem != null )
            pbElem.style.display = "block";

          var psElem = document.getElementById( "feedlySplashscreen" );
          if( psElem != null )
            psElem.style.display = "block";

          var wmElem = document.getElementById( "feedlyStaticProgressMessage" );
          if( wmElem != null )
          {
            var fc = getFeedlyCookie();
            if( fc != null )
            {
              wmElem.innerHTML = "Welcome back!";
            }
            else
            {
              wmElem.innerHTML = "Welcome!";
            }
          }
        }

        if( feedlyProgressCounter > feedlyProgressSteps )
        {
          window.clearInterval( feedlyProgressInterval );
          return;
        }

        var pbElem = document.getElementById( "feedlyStaticProgressBar" );
        if( pbElem == null )
          window.clearInterval( feedlyProgressInterval );
        else
          pbElem.style.width = parseInt ( feedlyProgressCounter / feedlyProgressSteps * window.innerWidth ) + "px";
      },
      parseInt( feedlyProgressDuration / feedlyProgressSteps )
    );
</script>
